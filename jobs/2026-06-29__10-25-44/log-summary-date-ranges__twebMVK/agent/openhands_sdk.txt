When using tools that support the security_risk parameter, assess the safety 
risk of your actions:

log_dir = Path('/app/logs')

]
severities = ['ERROR', 'WARNING', 'INFO']
counts = {period: {sev: 0 for sev in severities} for period, _, _ in periods}
counts['total'] = {sev: 0 for sev in severities}
sev_re = re.compile(r'\[(ERROR|WARNING|INFO)\]')

for path in sorted(log_dir.glob('*.log')):
    file_date = date.fromisoformat(path.name[:10])
    file_counts = {sev: 0 for sev in severities}
    with path.open('r', encoding='utf-8') as f:
        for line in f:
            match = sev_re.search(line)
            if match:
                file_counts[match.group(1)] += 1
    for sev, count in file_counts.items():
        counts['total'][sev] += count
    for period, start, end in periods:
        if start <= file_date <= end:
            for sev, count in file_counts.items():
                counts[period][sev] += count

with out_path.open('w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    writer.writerow(['period', 'severity', 'count'])
    for period, _, _ in periods:
        for sev in severities:
            writer.writerow([period, sev, counts[period][sev]])
    for sev in severities:
        writer.writerow(['total', sev, counts['total'][sev]])
